## 四、软件设计（改进版）

### 4.1 Goertzel算法（新增）

**作用**：从采样数据中提取单一频率的幅度和相位

```c
// Goertzel算法结构体
typedef struct {
    float coeff;     // 2*cos(2πk/N)
    float s0, s1, s2; // 状态变量
    int N;           // 采样点数
} Goertzel;

// 初始化
void Goertzel_Init(Goertzel *g, float target_freq, float sample_rate, int N) {
    g->N = N;
    int k = (int)(0.5 + N * target_freq / sample_rate);
    g->coeff = 2.0f * cosf(2.0f * PI * k / N);
    g->s0 = g->s1 = g->s2 = 0;
}

// 逐点处理
void Goertzel_Process(Goertzel *g, float sample) {
    g->s0 = sample + g->coeff * g->s1 - g->s2;
    g->s2 = g->s1;
    g->s1 = g->s0;
}

// 提取幅度和相位
void Goertzel_Result(Goertzel *g, float *magnitude, float *phase) {
    float real = g->s1 - g->s2 * cosf(2.0f * PI / g->N);
    float imag = g->s2 * sinf(2.0f * PI / g->N);
    
    *magnitude = 2.0f * sqrtf(real*real + imag*imag) / g->N;
    *phase = atan2f(imag, real) * 180.0f / PI;
}
```

**优势**：
- 计算量：N次乘加（N=采样点数），远小于FFT
- 同时输出幅度和相位
- 适合单频率提取，完美匹配本题需求

### 4.2 滤波类型识别（改进版）

```c
typedef enum {
    FILTER_LOWPASS = 0,
    FILTER_HIGHPASS,
    FILTER_BANDPASS,
    FILTER_BANDSTOP
} FilterType;

// 幅度+相位联合识别
FilterType IdentifyFilterType(FreqResponse *resp, int n) {
    float gain_low = resp[0].magnitude;           // 低频增益
    float gain_high = resp[n-1].magnitude;        // 高频增益
    float phase_low = resp[0].phase;              // 低频相位
    float phase_high = resp[n-1].phase;           // 高频相位
    
    // 找峰值和谷值
    float gain_max = -999, gain_min = 999;
    int max_idx = 0, min_idx = 0;
    for (int i = 0; i < n; i++) {
        if (resp[i].magnitude > gain_max) {
            gain_max = resp[i].magnitude;
            max_idx = i;
        }
        if (resp[i].magnitude < gain_min) {
            gain_min = resp[i].magnitude;
            min_idx = i;
        }
    }
    
    // 幅度判断
    int amp_type; // 0=低通, 1=高通, 2=带通, 3=带阻
    if (gain_low > gain_high + 6.0f && gain_low > gain_max - 3.0f) {
        amp_type = 0; // 低通
    } else if (gain_high > gain_low + 6.0f && gain_high > gain_max - 3.0f) {
        amp_type = 1; // 高通
    } else if (gain_max > gain_low + 6.0f && gain_max > gain_high + 6.0f) {
        amp_type = 2; // 带通
    } else if (gain_min < gain_low - 6.0f && gain_min < gain_high - 6.0f) {
        amp_type = 3; // 带阻
    }
    
    // 相位验证
    // 低通：相位从0°到-180°
    // 高通：相位从0°到+180°
    // 带通：相位从+90°到-90°（穿越0°）
    // 带阻：相位从0°跳变到±180°再回到0°
    
    return (FilterType)amp_type;
}
```

### 4.3 RLC参数解析拟合（改进版）

```c
// 从频率响应提取RLC参数
CircuitModel FitRLC_Analytical(FreqResponse *resp, int n, FilterType type) {
    CircuitModel model;
    model.type = type;
    
    // 找谐振频率（带通/带阻的中心频率，或低通/高通的截止频率）
    float f_resonant = 0, Q = 0, gain_peak = 0;
    
    if (type == FILTER_BANDPASS || type == FILTER_BANDSTOP) {
        // 带通/带阻：峰值/谷值频率就是谐振频率
        int peak_idx = 0;
        float peak_val = (type == FILTER_BANDPASS) ? -999 : 999;
        for (int i = 0; i < n; i++) {
            if ((type == FILTER_BANDPASS && resp[i].magnitude > peak_val) ||
                (type == FILTER_BANDSTOP && resp[i].magnitude < peak_val)) {
                peak_val = resp[i].magnitude;
                peak_idx = i;
            }
        }
        f_resonant = resp[peak_idx].freq;
        
        // 计算Q值（3dB带宽法）
        float bw = Find3dBBandwidth(resp, n, peak_val);
        Q = f_resonant / bw;
    } else {
        // 低通/高通：截止频率是-3dB点
        float dc_gain = resp[0].magnitude;
        float cutoff_gain = dc_gain - 3.0f;
        for (int i = 0; i < n; i++) {
            if (resp[i].magnitude <= cutoff_gain) {
                f_resonant = resp[i].freq;
                break;
            }
        }
        Q = 0.707; // Butterworth响应
    }
    
    // 从f0和Q计算L和C
    // f0 = 1/(2π√(LC)), Q = √(L/C)/R
    
    // 枚举L值，计算C和R
    float L_vals[] = {1e-3, 2.2e-3, 3.3e-3, 4.7e-3, 6.8e-3, 10e-3};
    float best_error = 1e9;
    
    for (int l = 0; l < 6; l++) {
        float L = L_vals[l];
        float C = 1.0f / (4 * PI * PI * f_resonant * f_resonant * L);
        
        // 验证C在范围内
        if (C >= 10e-9 && C <= 100e-9) {
            float R = sqrtf(L / C) / Q;
            
            // 验证R在范围内
            if (R >= 1000 && R <= 10000) {
                // 计算拟合误差
                float error = CalcFitError(R, L, C, resp, n);
                if (error < best_error) {
                    best_error = error;
                    model.R = R;
                    model.L = L;
                    model.C = C;
                }
            }
        }
    }
    
    return model;
}
```

### 4.4 PID自动微调（新增）

```c
// 上电自动阶跃测试微调PID参数
void PID_AutoTune(PID_Controller *pid) {
    // 1. 发送阶跃信号
    SetDDS_Amplitude(0.5f);  // 0.5V阶跃
    HAL_Delay(100);          // 等待稳态
    
    // 2. 采集阶跃响应
    float response[200];
    for (int i = 0; i < 200; i++) {
        response[i] = Measure_OutputVpp();
        HAL_Delay(1);
    }
    
    // 3. 提取特征参数
    float overshoot = CalcOvershoot(response, 200);
    float rise_time = CalcRiseTime(response, 200);
    float settling_time = CalcSettlingTime(response, 200);
    
    // 4. 根据特征调整PID参数
    if (overshoot > 10.0f) {
        // 超调过大，减小Kp，增大Kd
        pid->Kp *= 0.8f;
        pid->Kd *= 1.2f;
    } else if (rise_time > 50.0f) {
        // 响应过慢，增大Kp，减小Kd
        pid->Kp *= 1.2f;
        pid->Kd *= 0.8f;
    }
    
    // 限制积分项防止饱和
    if (pid->Ki > 0.1f) pid->Ki = 0.1f;
}
```
