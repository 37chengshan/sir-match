### 2.2 已知模型电路设计

**传递函数**：
$$H(s)=\frac{5}{10^{-8}s^2+3\times10^{-4}s+1}$$

**电路拓扑**：两级RC低通 + 同相放大

```
第一级一阶RC低通：
        R1 (10kΩ)
Vin ──┤├──────── V1
                 │
                C1 (10nF)
                 │
                GND

第二级一阶RC低通：
        R2 (15kΩ)
V1  ──┤├──────── V2
                 │
                C2 (6.8nF)
                 │
                GND

同相放大级（增益=5）：
                    Rf (40kΩ)
V2  ──(+)[OPA2134]──┤├──┬── Vout
        │               │
        └─── Rg (10kΩ)──┘
                │
               GND
```

**元件值计算**：
- 截止频率：fc = 1/(2π×R1×C1) = 1/(2π×10⁴×10⁻⁸) ≈ 1592 Hz
- 放大增益：K = 1 + Rf/Rg = 1 + 40k/10k = 5

**误差分析**：
- 电阻精度：1%金属膜电阻
- 电容精度：5%陶瓷电容
- 预期总误差：< 8%（满足≤10%要求）

### 3.2 核心算法

#### 3.2.1 PID自动幅度控制

```c
// PID控制器结构
typedef struct {
    float Kp;           // 比例系数
    float Ki;           // 积分系数
    float Kd;           // 微分系数
    float integral;     // 积分累积
    float prev_error;   // 上次误差
    float output_min;   // 输出下限
    float output_max;   // 输出上限
} PID_Controller;

// PID更新函数
float PID_Update(PID_Controller *pid, float setpoint, float measured) {
    float error = setpoint - measured;
    
    // 积分项（带抗饱和）
    pid->integral += error;
    if (pid->integral > 1000.0f) pid->integral = 1000.0f;
    if (pid->integral < -1000.0f) pid->integral = -1000.0f;
    
    // 微分项
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    
    // PID输出
    float output = pid->Kp * error 
                 + pid->Ki * pid->integral 
                 + pid->Kd * derivative;
    
    // 输出限幅
    if (output > pid->output_max) output = pid->output_max;
    if (output < pid->output_min) output = pid->output_min;
    
    return output;
}
```

#### 3.2.2 频率响应测量

```c
// 频率响应数据结构
typedef struct {
    float freq;         // 频率(Hz)
    float magnitude;    // 增益(dB)
    float phase;        // 相位(度)
} FreqResponse;

// 扫频测量函数
void SweepFrequency(FreqResponse *response, int num_points) {
    float freq_start = 100.0f;     // 起始频率
    float freq_end = 100000.0f;    // 终止频率
    
    for (int i = 0; i < num_points; i++) {
        // 对数间隔
        float freq = freq_start * powf(freq_end/freq_start, 
                                       (float)i/(num_points-1));
        
        // 设置DDS频率
        AD9833_SetFrequency(freq);
        AD9833_SetAmplitude(1.0f);  // 1Vpp
        
        // 等待稳态
        HAL_Delay(20);
        
        // 采集信号
        float vin = Measure_InputVpp();
        float vout = Measure_OutputVpp();
        float phase = Measure_PhaseDiff();
        
        // 记录数据
        response[i].freq = freq;
        response[i].magnitude = 20.0f * log10f(vout / vin);
        response[i].phase = phase;
    }
}
```

#### 3.2.3 滤波类型识别

```c
typedef enum {
    FILTER_LOWPASS = 0,
    FILTER_HIGHPASS,
    FILTER_BANDPASS,
    FILTER_BANDSTOP
} FilterType;

FilterType IdentifyFilterType(FreqResponse *resp, int n) {
    float gain_low = resp[0].magnitude;
    float gain_high = resp[n-1].magnitude;
    float gain_max = -999.0f;
    int max_idx = 0;
    
    // 找峰值
    for (int i = 0; i < n; i++) {
        if (resp[i].magnitude > gain_max) {
            gain_max = resp[i].magnitude;
            max_idx = i;
        }
    }
    
    // 判断类型
    if (gain_low > gain_high + 6.0f) {
        return FILTER_LOWPASS;
    } else if (gain_high > gain_low + 6.0f) {
        return FILTER_HIGHPASS;
    } else if (gain_max > gain_low + 6.0f && gain_max > gain_high + 6.0f) {
        return FILTER_BANDPASS;
    } else {
        return FILTER_BANDSTOP;
    }
}
```

#### 3.2.4 RLC参数拟合

```c
typedef struct {
    float R;            // 电阻(Ω)
    float L;            // 电感(H)
    float C;            // 电容(F)
    FilterType type;    // 滤波类型
} CircuitModel;

CircuitModel FitRLCParameters(FreqResponse *resp, int n, FilterType type) {
    CircuitModel model;
    model.type = type;
    
    // 提取关键参数
    float fc = FindCutoffFreq(resp, n);    // 截止/中心频率
    float bw = FindBandwidth(resp, n);     // 带宽
    float Q = fc / bw;                     // 品质因数
    
    // 枚举LC组合，计算R
    float L_vals[] = {1e-3, 2.2e-3, 3.3e-3, 4.7e-3, 6.8e-3, 10e-3};
    float C_vals[] = {10e-9, 22e-9, 33e-9, 47e-9, 68e-9, 100e-9};
    
    float best_error = 1e9f;
    
    for (int l = 0; l < 6; l++) {
        for (int c = 0; c < 6; c++) {
            float L = L_vals[l];
            float C = C_vals[c];
            
            // 计算理论R值
            float R_theory = sqrtf(L / C) / Q;
            
            // 验证范围
            if (R_theory >= 1000.0f && R_theory <= 10000.0f) {
                // 计算拟合误差
                float error = CalcFitError(R_theory, L, C, resp, n);
                
                if (error < best_error) {
                    best_error = error;
                    model.R = R_theory;
                    model.L = L;
                    model.C = C;
                }
            }
        }
    }
    
    return model;
}
```

#### 3.2.5 逆滤波器设计与波形复刻

```c
// 逆滤波器结构
typedef struct {
    float b[3];     // 分子系数
    float a[3];     // 分母系数
    float w[2];     // 状态变量
} InvFilter;

// 设计逆滤波器
InvFilter DesignInvFilter(CircuitModel model, float fs) {
    InvFilter filt;
    
    float LC = model.L * model.C;
    float RC = model.R * model.C;
    float T = 1.0f / fs;
    float K = 2.0f / T;
    float K2 = K * K;
    
    // 双线性变换
    if (model.type == FILTER_LOWPASS) {
        // H(s) = 1/(LCs²+RCs+1)
        // H_inv(s) = LCs²+RCs+1
        filt.b[0] = LC*K2 + RC*K + 1.0f;
        filt.b[1] = 2.0f * (1.0f - LC*K2);
        filt.b[2] = LC*K2 - RC*K + 1.0f;
        filt.a[0] = 1.0f;
        filt.a[1] = 0.0f;
        filt.a[2] = 0.0f;
    }
    // ... 其他滤波类型
    
    // 归一化
    float a0 = filt.a[0];
    for (int i = 0; i < 3; i++) {
        filt.b[i] /= a0;
        filt.a[i] /= a0;
    }
    
    filt.w[0] = 0.0f;
    filt.w[1] = 0.0f;
    
    return filt;
}

// 滤波执行
float InvFilter_Execute(InvFilter *f, float input) {
    float w = input - f->a[1]*f->w[0] - f->a[2]*f->w[1];
    float output = f->b[0]*w + f->b[1]*f->w[0] + f->b[2]*f->w[1];
    
    f->w[1] = f->w[0];
    f->w[0] = w;
    
    return output;
}
```
